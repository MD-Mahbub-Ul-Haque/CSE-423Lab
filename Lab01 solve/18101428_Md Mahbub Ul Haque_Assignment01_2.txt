		   bf = new BufferedReader(flr);
		   gl.glBegin (GL2.GL_POINTS);
		   while((st = bf.readLine()) != null && !st.equals("")) {
			  String[] c = st.split(" ");
			  float x = Float.parseFloat(c[0]);
			  float y = Float.parseFloat(c[1]);
			  System.out.println(i+". "+"x: "+x+" y: "+y); 
			  i++;
			  gl.glVertex2d(x,y);
		   }
		   gl.glEnd(); 
	   }catch(IOException e) {

	   }
	   
	  }

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Random;
public class coordinate {

	public static void main(String[] args) {
		
		BufferedWriter bf;
		Random rand = new Random();
		String coordinates = "";
		try {
			for(int i = 0; i < 20; i++) {
				float x = -1.0f+(1.0f-(-1.0f))*rand.nextFloat();
				float y = -1.0f+(1.0f-(-1.0f))*rand.nextFloat();
				coordinates += x+" "+y+"\n";
			}
			
			File file = new File("coordinates.txt");
				
			if(!file.exists()) {
				file.createNewFile();
			}
			
			FileWriter fr = new FileWriter(file);
			bf = new BufferedWriter(fr);
			bf.flush();
			bf.write(coordinates);
			bf.close();
				
		}catch(IOException e){
			e.printStackTrace();
		}
		

	}

}
